package com.cms.service.practice4;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.thymeleaf.util.StringUtils;

import com.cms.entity.marketstore.MarketStoreBean;
import com.cms.form.practice4.ReadDataBaseForm;
import com.cms.mapper.practice4.Practice4Mapper;
import com.exception.BusinessException;

/**
 * 社員情報取得サービス
 */
@Service
public class ReadDataBaseServiceImpl1 implements ReadDataBaseService1 {

	@Autowired
	Practice4Mapper mapper;

	/**
	 * 社員情報を検索する
	 * 
	 * @param form 社員検索画面フォーム
	 * @return 社員Beanリスト
	 * @throws Exception 
	 */	
	@Override
	public List<MarketStoreBean> readDataFromMysql(ReadDataBaseForm form) throws Exception {
		
		MarketStoreBean param = new MarketStoreBean();
		if (!StringUtils.isEmpty(form.getStoreId()))	{
			param.setStoreId(form.getStoreId());
		}
		
		if (!StringUtils.isEmpty(form.getStoreName()))	{
			param.setStoreName(form.getStoreName());
		}
		List<MarketStoreBean> results = mapper.select(param);

//		for(CmsEmployeeBean b : results) {
//			b.setAddress("aaaaaaa");
//		}
		if (results.size() == 0) {
			throw new BusinessException("検索結果がありません。");
		}
		
		return results;
	}


}
