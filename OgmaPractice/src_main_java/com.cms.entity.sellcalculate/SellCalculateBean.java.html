package com.cms.entity.sellcalculate;

import java.sql.Timestamp;

import javax.persistence.Column;
import javax.persistence.Id;
import javax.persistence.Table;

import lombok.Data;

/**
 * 請求書Bean
 */

@Data
@Table(name = "market_benefit")
public class SellCalculateBean{
	
	@Id
	@Column(name = "benefitId")
	private String benefitId;

	@Column(name = "storeId")
	private String storeId;
	
	@Column(name = "salesCount")
	private int salesCount;
	
	@Column(name = "salesAmount")
	private double salesAmount;
	
	@Column(name = "benefit")
	private double benefit;
	
	@Column(name = "registDay")
	private Timestamp registDay;
	
	@Column(name = "updateDay")
	private Timestamp updateDay;
	
	
}
