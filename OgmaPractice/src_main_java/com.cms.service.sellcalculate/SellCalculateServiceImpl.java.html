package com.cms.service.sellcalculate;

import java.sql.Timestamp;
import java.util.ArrayList;
import java.util.Calendar;
import java.util.Date;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.dao.DuplicateKeyException;
import org.springframework.stereotype.Service;

import com.cms.entity.marketstore.MarketStoreBean;
import com.cms.entity.sellcalculate.SellDataBean;
import com.cms.entity.sellcsv.SellCsvBean;
import com.cms.mapper.sellcalculate.SellCalculateMapper;
import com.exception.BusinessException;

import io.micrometer.core.instrument.util.StringUtils;

/**
 * 請求書計算ServiceImpl
 */
@SuppressWarnings("rawtypes")
@Service
public class SellCalculateServiceImpl implements SellCalculateService {

	@Autowired
	SellCalculateMapper mapper;
	private MarketStoreBean MarketStoreBean;
	private List<SellCsvBean> sellCsvList;
	private List<SellDataBean> registList;

	/**
	 * 請求書データを登録する
	 * 
	 * @param employeeId 社員ID
	 * @param workMonth  勤務年月
	 * @return List<SellDataBean> 計算後の請求書
	 */
	@Override
	public List<SellDataBean> calculateSellData(String storeId, String sellMonth) throws Exception {

		// 変数の初期化
		sellCsvList = new ArrayList<SellCsvBean>();
		registList = new ArrayList<SellDataBean>();

		// チェック処理
		check(storeId, sellMonth);

		// 勤務時間を統計する
		calculate();

		// 請求書の登録
		registeData();

		// 登録結果を戻す
		
		return registList;
	}

	/**
	 * チェック処理
	 * 
	 * @param employeeId 社員ID
	 * @param workMonth  勤務年月
	 * 
	 * @throws Exception
	 */
	private void check(String storeId, String sellMonth) throws Exception {

		// ■入力チェック
		if (StringUtils.isEmpty(storeId)) {
			// 社員の必須チェック
			throw new BusinessException("IDをご設定ください");
		}

		// ■8文字以内の文字列のチェック
		if (storeId.length() > 8) {
			// 社員の必須チェック
			throw new BusinessException("8文字以内のIDをご入力ください。");
		}

		// ■マスタ存在チェック（社員情報）
		MarketStoreBean inputBean = new MarketStoreBean();
		inputBean.setStoreId(storeId);
		MarketStoreBean = mapper.selectStore(inputBean);
		if (MarketStoreBean == null) {
			throw new BusinessException("店舗は存在しません。");
		}

		// ■マスタ存在チェック（卖上）
		SellCsvBean sellInfo = new SellCsvBean();
		sellInfo.setStoreId(storeId);
		if (!StringUtils.isEmpty(sellMonth)) {
			sellInfo.setSalesDay(sellMonth);
		}
		sellCsvList = mapper.selectMarketSales(sellInfo);
		if (sellCsvList == null || sellCsvList.size() == 0) {
			throw new BusinessException("売上情報は存在しません。");
		}
	}

	/**
	 * 勤務時間計算
	 * 
	 */
	private void calculate() {

		// 月毎に勤務情報の1件目を保存する（共通項目を利用するため）

		// 出勤時間の合計をMapに保存する
//		Map<String, Integer> workHourMap = new HashMap<String, Integer>();

		// 2.1システム日付の取得
		Date today = Calendar.getInstance().getTime();
		Timestamp timestamp = new Timestamp(today.getTime());

		String key = "";
		int salesCount = 0;
		double salesAmount = 0;
		double cost = 0;
		double benefit = 0;
		String storeId = "";
		String salesDay = "";
		String maxId = mapper.selectMaxId();
		int recordId = Integer.valueOf(maxId) + 1;

		// ■勤務時間の合計
		for (SellCsvBean bean : sellCsvList) {
			if (key != bean.getStoreId() + bean.getSalesDay()) {
				// グループが違う場合
				key = bean.getStoreId() + bean.getSalesDay();
				// 2.2.3レコードの作成
				if (salesCount != 0) {
					SellDataBean calGroup = new SellDataBean();
					String maxBenefitId = String.valueOf(recordId);
					calGroup.setBenefitId(maxBenefitId);
					calGroup.setSalesCount(salesCount);
					calGroup.setSalesAmount(salesAmount);
					calGroup.setCost(cost);
					calGroup.setBenefit(benefit);
					calGroup.setStoreId(storeId);
					calGroup.setSalesDay(salesDay);
					calGroup.setRegistDay(timestamp);
					calGroup.setUpdateDay(timestamp);
					registList.add(calGroup);
					recordId ++;

					// 2.2.4合計項目クリア
					salesCount = 0;
					salesAmount = 0;
					cost = 0;
					benefit = 0;
				}
				// 2.2.2項目合計
				// 合計売買件数＝合計売買件数＋【1.3売上検索】・購入件数
				salesCount = bean.getSalesCount();
				// 合計売買金額＝合計売買金額+【1.3売上検索】・購入件数＊【1.3売上検索】・販売単価
				salesAmount = bean.getSalesCount() * Double.parseDouble(bean.getSalesPrice());
				// 合計仕入金額＝合計仕入金額+【1.3売上検索】・購入件数＊【1.3売上検索】・仕入単価
				cost = bean.getSalesCount() * Double.parseDouble(bean.getPurchasePrice());
				// 合計利益＝合計利益+【1.3売上検索】・購入件数＊（【1.3売上検索】・販売単価-【1.3売上検索】・仕入単価）
				benefit = bean.getSalesCount()
						* (Double.parseDouble(bean.getSalesPrice()) - Double.parseDouble(bean.getPurchasePrice()));
				storeId = bean.getStoreId();
				salesDay = bean.getSalesDay();

			} else if (key == bean.getStoreId() + bean.getSalesDay()) {
				salesCount += bean.getSalesCount();
				salesAmount += bean.getSalesCount() * Double.parseDouble(bean.getSalesPrice());
				cost += bean.getSalesCount() * Double.parseDouble(bean.getPurchasePrice());
				benefit += bean.getSalesCount()
						* (Double.parseDouble(bean.getSalesPrice()) - Double.parseDouble(bean.getPurchasePrice()));

			}

		}
		SellDataBean calGroup = new SellDataBean();
		String maxBenefitId = String.valueOf(recordId);
		calGroup.setBenefitId(maxBenefitId);
		calGroup.setSalesCount(salesCount);
		calGroup.setSalesAmount(salesAmount);
		calGroup.setCost(cost);
		calGroup.setBenefit(benefit);
		calGroup.setStoreId(storeId);
		calGroup.setSalesDay(salesDay);
		calGroup.setRegistDay(timestamp);
		calGroup.setUpdateDay(timestamp);
		registList.add(calGroup);
		recordId++;
		}

	// 作成されるレコードを【月次売買(販売店)登録リスト】に保存する

	/**
	 * 請求書の登録
	 * 
	 */
	private void registeData() {
		try {
			mapper.insertRecords(registList);
		} catch (DuplicateKeyException e1) {
			// 登録キー重複エラー
			throw new BusinessException("指定条件の請求書がすでに登録されました。", e1);
		} catch (Exception se) {
			// 予想外エラー
			throw new BusinessException(se.getMessage());
		}
	}
}
