		try {
			ReadDataBaseForm form = new ReadDataBaseForm();
			form.setStoreId("00000003");
			form.setStoreName("浅草店");
			List<MarketStoreBean> results = service.readDataFromMysql(form);

			//検査結果のサイズ
			assertEquals(1, results.size());
			for (MarketStoreBean bean : results) {
				//検索明細の比較
				assertEquals("00000003", bean.getStoreId());
				assertEquals("浅草店", bean.getStoreName());
				assertEquals("东京都大草区洼洼村", bean.getAddress());
				assertEquals("08035451503", bean.getPhone());
				assertEquals("2023-05-03", bean.getStartDay());
				assertEquals("2023-05-13", bean.getFinishDay());
			}
			
		} catch (Exception e) {
			System.out.println(e.getMessage());
		}
	}
